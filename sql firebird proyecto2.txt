/* =============================================================================
   2. GEOGRAFÍA
   ============================================================================= */

CREATE TABLE Pais (
    id_pais        INTEGER NOT NULL PRIMARY KEY,
    nombre_pais    VARCHAR(100) NOT NULL,
    codigo_iso     VARCHAR(10)
);

CREATE TABLE Region (
    id_region      INTEGER NOT NULL PRIMARY KEY,
    id_pais        INTEGER NOT NULL,
    nombre_region  VARCHAR(100) NOT NULL,
    codigo_region  VARCHAR(10),
    CONSTRAINT fk_region_pais FOREIGN KEY (id_pais) REFERENCES Pais(id_pais)
);

CREATE TABLE Departamento (
    id_departamento     INTEGER NOT NULL PRIMARY KEY,
    id_region           INTEGER NOT NULL,
    nombre_departamento VARCHAR(100) NOT NULL,
    codigo_departamento VARCHAR(10),
    cabecera            VARCHAR(100),
    CONSTRAINT fk_depto_region FOREIGN KEY (id_region) REFERENCES Region(id_region)
);

CREATE TABLE Area_Geografica (
    id_area_geo     INTEGER NOT NULL PRIMARY KEY,
    nombre_area     VARCHAR(50) NOT NULL
);

CREATE TABLE Municipio (
    id_municipio        INTEGER NOT NULL PRIMARY KEY,
    id_departamento     INTEGER NOT NULL,
    id_area_geo         INTEGER,
    nombre_municipio    VARCHAR(100) NOT NULL,
    codigo_municipio    VARCHAR(10),
    poblacion_estimada  INTEGER,
    latitud              DECIMAL(10,6),
    longitud            DECIMAL(10,6),
    CONSTRAINT fk_muni_depto FOREIGN KEY (id_departamento) REFERENCES Departamento(id_departamento),
    CONSTRAINT fk_muni_area  FOREIGN KEY (id_area_geo) REFERENCES Area_Geografica(id_area_geo)
);

CREATE TABLE Persona (
    id_persona            INTEGER NOT NULL PRIMARY KEY,
    nombres               VARCHAR(100),
    apellidos             VARCHAR(100),
    fecha_nacimiento      DATE,
    id_sexo               INTEGER,
    id_etnia              INTEGER,
    id_estado_civil       INTEGER,
    id_religion           INTEGER,
    numero_identificacion VARCHAR(50),
    es_anonimo            SMALLINT DEFAULT 0,
    CONSTRAINT fk_pers_sexo  FOREIGN KEY (id_sexo) REFERENCES Sexo(id_sexo),
    CONSTRAINT fk_pers_etnia FOREIGN KEY (id_etnia) REFERENCES Etnia(id_etnia),
    CONSTRAINT fk_pers_ec    FOREIGN KEY (id_estado_civil) REFERENCES Estado_Civil(id_estado_civil),
    CONSTRAINT fk_pers_rel   FOREIGN KEY (id_religion) REFERENCES Religion(id_religion)
);

CREATE TABLE Persona_Idioma (
    id_persona_idioma INTEGER NOT NULL PRIMARY KEY,

    CONSTRAINT fk_pi_pers    FOREIGN KEY (id_persona) REFERENCES Persona(id_persona),
    CONSTRAINT fk_pi_idioma FOREIGN KEY (id_idioma) REFERENCES Idioma(id_idioma)
);

CREATE TABLE Persona_Discapacidad (
    id_persona_disc  INTEGER NOT NULL PRIMARY KEY,

);

    id_domicilio     INTEGER NOT NULL PRIMARY KEY,
    id_persona       INTEGER NOT NULL,
    id_municipio     INTEGER NOT NULL,

CREATE TABLE Escolaridad_Persona (
    id_escolaridad     INTEGER NOT NULL PRIMARY KEY,
    id_persona         INTEGER NOT NULL,

    CONSTRAINT fk_esc_pers_base  FOREIGN KEY (id_persona) REFERENCES Persona(id_persona),

    CONSTRAINT fk_esc_ie    FOREIGN KEY (id_institucion_edu) REFERENCES Institucion_Educativa(id_institucion_edu)

CREATE TABLE Ocupacion (
    id_ocupacion    INTEGER NOT NULL PRIMARY KEY,
    nombre_ocupacion VARCHAR(150) NOT NULL,
    codigo_ciuo      VARCHAR(10)
);

CREATE TABLE Tipo_Empleo (
    id_tipo_empleo   INTEGER NOT NULL PRIMARY KEY,
    nombre_tipo      VARCHAR(100) NOT NULL

CREATE TABLE Condicion_Laboral (
    id_cond_laboral     INTEGER NOT NULL PRIMARY KEY,
    nombre_cond_laboral VARCHAR(60) NOT NULL,

CREATE TABLE Empleo_Persona (
    id_empleo        INTEGER NOT NULL PRIMARY KEY,
    id_persona       INTEGER NOT NULL,
    id_ocupacion     INTEGER,
    id_tipo_empleo   INTEGER,
    id_cond_laboral  INTEGER,
    id_sector_econ   INTEGER,
    salario_mensual  DECIMAL(12,2),
    fecha_inicio     DATE,
    fecha_fin        DATE,
    es_actual        SMALLINT DEFAULT 1,
    CONSTRAINT fk_emp_pers_base   FOREIGN KEY (id_persona) REFERENCES Persona(id_persona),
    CONSTRAINT fk_emp_ocup   FOREIGN KEY (id_ocupacion) REFERENCES Ocupacion(id_ocupacion),
    CONSTRAINT fk_emp_tipo   FOREIGN KEY (id_tipo_empleo) REFERENCES Tipo_Empleo(id_tipo_empleo),
    CONSTRAINT fk_emp_cond   FOREIGN KEY (id_cond_laboral) REFERENCES Condicion_Laboral(id_cond_laboral),
    CONSTRAINT fk_emp_sector FOREIGN KEY (id_sector_econ) REFERENCES Sector_Economico(id_sector_econ)

CREATE TABLE Hogar (
    id_hogar              INTEGER NOT NULL PRIMARY KEY,

    tiene_electricidad    SMALLINT DEFAULT 0,

CREATE TABLE Persona_Hogar (
    id_persona_hogar INTEGER NOT NULL PRIMARY KEY,

CREATE TABLE Movil_Delito (

CREATE TABLE Hecho_Movil (

    CONSTRAINT fk_lab_muni FOREIGN KEY (id_municipio) REFERENCES Municipio(id_municipio)

CREATE TABLE Forense (
    id_forense   INTEGER NOT NULL PRIMARY KEY,

    numero_colegiado VARCHAR(30),
    especialidad   VARCHAR(100),
    CONSTRAINT fk_for_lab  FOREIGN KEY (id_lab_forense) REFERENCES Laboratorio_Forense(id_lab_forense),
    CONSTRAINT fk_for_pers_base FOREIGN KEY (id_persona) REFERENCES Persona(id_persona)
);

CREATE TABLE Necropsia (
    id_necropsia   INTEGER NOT NULL PRIMARY KEY,
    id_hecho       INTEGER NOT NULL,
    id_victima     INTEGER,
    id_forense     INTEGER,
    id_lab_forense INTEGER,
    fecha_necropsia DATE,
    causa_muerte   VARCHAR(250),
    hallazgos      BLOB SUB_TYPE TEXT,
    CONSTRAINT fk_nec_hecho FOREIGN KEY (id_hecho) REFERENCES Hecho_Delictivo(id_hecho),
    CONSTRAINT fk_nec_vic   FOREIGN KEY (id_victima) REFERENCES Victima(id_victima),
    CONSTRAINT fk_nec_for   FOREIGN KEY (id_forense) REFERENCES Forense(id_forense),
    CONSTRAINT fk_nec_lab   FOREIGN KEY (id_lab_forense) REFERENCES Laboratorio_Forense(id_lab_forense)
);

CREATE TABLE Exhumacion (
    id_exhumacion   INTEGER NOT NULL PRIMARY KEY,
    id_necropsia    INTEGER,
    id_forense      INTEGER,
    id_municipio    INTEGER,
    fecha_exhumacion DATE,
    motivo          VARCHAR(300),
    resultado       BLOB SUB_TYPE TEXT,
    CONSTRAINT fk_exh_nec  FOREIGN KEY (id_necropsia) REFERENCES Necropsia(id_necropsia),
    CONSTRAINT fk_exh_for  FOREIGN KEY (id_forense) REFERENCES Forense(id_forense),
    CONSTRAINT fk_exh_muni FOREIGN KEY (id_municipio) REFERENCES Municipio(id_municipio)
);

CREATE TABLE Juzgado (
    id_juzgado     INTEGER NOT NULL PRIMARY KEY,
    id_municipio   INTEGER,
    nombre_juzgado VARCHAR(150) NOT NULL,
    tipo_juzgado   VARCHAR(80),
    CONSTRAINT fk_jz_muni FOREIGN KEY (id_municipio) REFERENCES Municipio(id_municipio)
);

CREATE TABLE Tipo_Delito_Juzgado (
    id_tipo_delito_jz INTEGER NOT NULL PRIMARY KEY,
    nombre_tipo       VARCHAR(200) NOT NULL,
    articulo_legal    VARCHAR(100)
);

CREATE TABLE Pena (
    id_pena     INTEGER NOT NULL PRIMARY KEY,
    nombre_pena VARCHAR(100) NOT NULL
);

CREATE TABLE Sentencia (
    id_sentencia       INTEGER NOT NULL PRIMARY KEY,
    id_juzgado         INTEGER NOT NULL,
    id_tipo_delito_jz  INTEGER NOT NULL,
    id_pena            INTEGER,
    id_hecho           INTEGER,
    fecha_sentencia    DATE NOT NULL,
    anios_prision      INTEGER,
    monto_multa        DECIMAL(12,2),
    es_condenatoria    SMALLINT DEFAULT 1,
    es_firme           SMALLINT DEFAULT 0,
    CONSTRAINT fk_sen_jz    FOREIGN KEY (id_juzgado) REFERENCES Juzgado(id_juzgado),
    CONSTRAINT fk_sen_tipo  FOREIGN KEY (id_tipo_delito_jz) REFERENCES Tipo_Delito_Juzgado(id_tipo_delito_jz),
    CONSTRAINT fk_sen_pena  FOREIGN KEY (id_pena) REFERENCES Pena(id_pena),
    CONSTRAINT fk_sen_hecho FOREIGN KEY (id_hecho) REFERENCES Hecho_Delictivo(id_hecho)
